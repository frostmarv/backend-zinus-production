// Di master-data.controller.js atau sejenisnya
// GET /api/master-data/remain-quantity
async getRemainQuantity(req, res) {
  const { customerPo, sku, sCode } = req.query;
  
  try {
    // 1. Get quantityOrder dari master data
    const masterData = await db.query(
      'SELECT quantity_order FROM master_data WHERE customer_po = ? AND sku = ?',
      [customerPo, sku]
    );
    const quantityOrder = masterData[0]?.quantity_order || 0;
    
    // 2. Hitung total produksi actual dari database untuk layer ini
    const result = await db.query(
      'SELECT SUM(quantity_produksi) as total FROM production_cutting_entries WHERE customer_po = ? AND sku = ? AND s_code = ?',
      [customerPo, sku, sCode]
    );
    const totalProduced = result[0]?.total || 0;
    
    // 3. Hitung remain
    const remainQuantity = quantityOrder - totalProduced;
    
    return res.json({
      quantityOrder,
      totalProduced,
      remainQuantity
    });
  } catch (error) {
    console.error('Error getting remain quantity:', error);
    return res.status(500).json({ error: 'Failed to get remain quantity' });
  }
}
